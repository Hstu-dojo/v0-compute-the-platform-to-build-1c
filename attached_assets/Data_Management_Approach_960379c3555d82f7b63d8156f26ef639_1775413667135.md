
## Group 3: Usage & Token Ledger

This group is append-only. No rows are ever updated or deleted. The balance is always computed from the sum.

```mermaid
erDiagram
    token_ledger {
        UUID id PK
        UUID user_id FK
        ENUM event_type
        INTEGER amount
        UUID job_id FK
        UUID invoice_id FK
        VARCHAR task_type
        VARCHAR model_name
        VARCHAR provider
        INTEGER raw_provider_tokens
        DECIMAL multiplier
        INTEGER computed_credits
        VARCHAR idempotency_key
        TEXT note
        TIMESTAMPTZ created_at
    }

    token_balance_snapshots {
        UUID id PK
        UUID user_id FK
        INTEGER balance_at_snapshot
        INTEGER period_credits_used
        INTEGER period_credits_granted
        DATE snapshot_date
        TIMESTAMPTZ created_at
    }

    monthly_credit_grants {
        UUID id PK
        UUID user_id FK
        UUID subscription_id FK
        INTEGER credits_granted
        DATE billing_period_start
        DATE billing_period_end
        BOOLEAN processed
        TIMESTAMPTZ processed_at
        TIMESTAMPTZ created_at
    }

    model_config {
        UUID id PK
        VARCHAR task_type
        VARCHAR provider
        VARCHAR model_name
        VARCHAR model_version
        DECIMAL token_multiplier
        INTEGER rollout_percentage
        VARCHAR fallback_model_name
        VARCHAR fallback_provider
        BOOLEAN is_active
        TIMESTAMPTZ rolled_out_at
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    usage_stats {
        UUID id PK
        UUID user_id FK
        DATE stat_date
        INTEGER total_jobs_run
        INTEGER total_credits_used
        JSONB jobs_by_type
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    users ||--o{ token_ledger : tracks
    users ||--o{ token_balance_snapshots : summarizes
    users ||--o{ monthly_credit_grants : receives
    subscriptions ||--o{ monthly_credit_grants : provides
    users ||--o{ usage_stats : aggregates
```

## Group 7: Workers & Job Management

This covers BullMQ job state tracking, retry history, and worker health monitoring.

```mermaid
erDiagram
    bullmq_jobs {
        UUID id PK
        UUID user_id FK
        UUID batch_id FK
        VARCHAR queue_name
        VARCHAR task_type
        VARCHAR bullmq_job_id
        JSONB input_payload
        ENUM status
        INTEGER attempt_number
        INTEGER max_attempts
        TEXT error_message
        JSONB error_stack
        DECIMAL processing_time_ms
        TIMESTAMPTZ queued_at
        TIMESTAMPTZ started_at
        TIMESTAMPTZ completed_at
        TIMESTAMPTZ next_retry_at
        TIMESTAMPTZ created_at
    }

    job_retry_history {
        UUID id PK
        UUID job_id FK
        INTEGER attempt_number
        ENUM status
        TEXT error_message
        VARCHAR provider_used
        VARCHAR model_used
        DECIMAL latency_ms
        TIMESTAMPTZ attempted_at
    }

    worker_health {
        UUID id PK
        VARCHAR worker_name
        VARCHAR queue_name
        VARCHAR host
        INTEGER pid
        ENUM status
        INTEGER jobs_processed_total
        INTEGER jobs_failed_total
        DECIMAL avg_processing_time_ms
        TIMESTAMPTZ last_heartbeat_at
        TIMESTAMPTZ started_at
    }

    dead_letter_jobs {
        UUID id PK
        UUID original_job_id FK
        UUID user_id FK
        VARCHAR queue_name
        VARCHAR task_type
        JSONB input_payload
        TEXT final_error
        INTEGER total_attempts
        BOOLEAN credits_released
        BOOLEAN requires_manual_review
        TIMESTAMPTZ created_at
    }

    job_events {
        UUID id PK
        UUID job_id FK
        ENUM event_type
        JSONB metadata
        TIMESTAMPTZ created_at
    }

    bullmq_jobs ||--o{ job_retry_history : tracks
    bullmq_jobs ||--o{ job_events : emits
    study_set_batches ||--o{ bullmq_jobs : belongs_to
    bullmq_jobs ||--o{ dead_letter_jobs : fails_into
```
