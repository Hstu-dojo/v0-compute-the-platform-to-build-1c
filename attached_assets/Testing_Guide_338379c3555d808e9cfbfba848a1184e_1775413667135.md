### `tests/helpers/request.helper.ts`

```tsx
import { FastifyInstance } from 'fastify'
import { mockFirebaseToken } from './auth.helper.js'

export function makeRequest(app: FastifyInstance) {
  return {
    get: (url: string, userId?: string) =>
      app.inject({
        method: 'GET',
        url: `/api/v1${url}`,
        headers: userId ? { authorization: `Bearer ${mockFirebaseToken(userId)}` } : {},
      }),

    post: (url: string, body: unknown, userId?: string) =>
      app.inject({
        method: 'POST',
        url: `/api/v1${url}`,
        headers: {
          'content-type': 'application/json',
          ...(userId ? { authorization: `Bearer ${mockFirebaseToken(userId)}` } : {}),
        },
        payload: JSON.stringify(body),
      }),

    delete: (url: string, userId?: string) =>
      app.inject({
        method: 'DELETE',
        url: `/api/v1${url}`,
        headers: userId ? { authorization: `Bearer ${mockFirebaseToken(userId)}` } : {},
      }),
  }
}
```

### `tests/integration/auth.test.ts`

```tsx
import { describe, it, expect, beforeAll, afterAll, afterEach } from 'vitest'
import { buildApp } from '../../src/app.js'
import { FastifyInstance } from 'fastify'
import { makeRequest } from '../helpers/request.helper.js'
import { teardown } from '../setup.js'
import { setupFirebaseMock } from '../helpers/auth.helper.js'
import { db } from '../../src/db/index.js'
import { users } from '../../src/db/schema/index.js'
import { eq } from 'drizzle-orm'

setupFirebaseMock()

describe('Auth routes', () => {
  let app: FastifyInstance
  let request: ReturnType<typeof makeRequest>

  beforeAll(async () => {
    app = await buildApp()
    request = makeRequest(app)
  })

  afterAll(async () => {
    await app.close()
  })

  afterEach(async () => {
    await teardown()
  })

  describe('POST /auth/register', () => {
    it('creates a new user, free subscription, and initial credit grant', async () => {
      const uid = 'test_register_uid'
      const response = await request.post('/auth/register', { display_name: 'Test Student' }, uid)

      expect(response.statusCode).toBe(201)

      const body = JSON.parse(response.body)
      expect(body.user.firebase_uid).toBe(uid)
      expect(body.user.role).toBe('student')
      expect(body.subscription.plan).toBe('free')
      expect(body.token_balance).toBeGreaterThan(0)
    })

    it('returns 409 if user already exists', async () => {
      const uid = 'duplicate_uid'
      await request.post('/auth/register', { display_name: 'First' }, uid)
      const second = await request.post('/auth/register', { display_name: 'Second' }, uid)

      expect(second.statusCode).toBe(409)
      const body = JSON.parse(second.body)
      expect(body.error.code).toBe('USER_ALREADY_EXISTS')
    })

    it('returns 401 with no auth token', async () => {
      const response = await request.post('/auth/register', { display_name: 'No Token' })
      expect(response.statusCode).toBe(401)
    })
  })

  describe('POST /auth/session', () => {
    it('returns user, session, subscription, and token_balance', async () => {
      const uid = 'session_test_uid'
      await request.post('/auth/register', { display_name: 'Session Test' }, uid)

      const response = await request.post('/auth/session', { device_type: 'web', device_name: 'Chrome' }, uid)

      expect(response.statusCode).toBe(200)
      const body = JSON.parse(response.body)
      expect(body.user.firebase_uid).toBe(uid)
      expect(body.session.session_id).toBeDefined()
      expect(body.token_balance).toBeTypeOf('number')
    })
  })
})
```

```tsx
import { describe, it, expect, beforeAll, afterAll, afterEach, vi } from 'vitest'
import { buildApp } from '../../src/app.js'
import { FastifyInstance } from 'fastify'
import { makeRequest } from '../helpers/request.helper.js'
import { teardown } from '../setup.js'
import { setupFirebaseMock } from '../helpers/auth.helper.js'
import { createTestUser, createTestDocument, createTestModelConfig, createTestPromptTemplate, grantTestCredits, createTestPlan, createTestSubscription } from '../helpers/db.helper.js'

setupFirebaseMock()

vi.mock('../../src/queues/index.js', () => ({
  studySetsQueue: {
    notes: { add: vi.fn().mockResolvedValue({ id: 'bullmq_mock_id' }) },
    flashcards: { add: vi.fn().mockResolvedValue({ id: 'bullmq_mock_id' }) },
  },
}))

describe('Study sets routes', () => {
  let app: FastifyInstance
  let request: ReturnType<typeof makeRequest>

  beforeAll(async () => {
    app = await buildApp()
    request = makeRequest(app)
  })

  afterAll(async () => {
    await app.close()
  })

  afterEach(async () => {
    await teardown()
  })

  describe('POST /study-sets/generate', () => {
    it('returns 202 with batch_id and job list when document is ready and balance is sufficient', async () => {
      const user = await createTestUser()
      const plan = await createTestPlan()
      await createTestSubscription(user.id, plan.id)
      await grantTestCredits(user.id, 2000)
      const doc = await createTestDocument(user.id)
      await createTestModelConfig('notes')
      await createTestModelConfig('flashcards')
      await createTestPromptTemplate('notes')

      const response = await request.post(
        '/study-sets/generate',
        { document_id: doc.id, types: ['notes', 'flashcards'] },
        user.firebaseUid
      )

      expect(response.statusCode).toBe(202)
      const body = JSON.parse(response.body)
      expect(body.batch.status).toBe('processing')
      expect(body.jobs).toHaveLength(2)
      expect(body.websocket.url).toContain('wss://')
    })

    it('returns 402 when token balance is insufficient', async () => {
      const user = await createTestUser()
      const plan = await createTestPlan()
      await createTestSubscription(user.id, plan.id)
      await grantTestCredits(user.id, 10)
      const doc = await createTestDocument(user.id)
      await createTestModelConfig('notes')

      const response = await request.post(
        '/study-sets/generate',
        { document_id: doc.id, types: ['notes'] },
        user.firebaseUid
      )

      expect(response.statusCode).toBe(402)
      const body = JSON.parse(response.body)
      expect(body.error.code).toBe('INSUFFICIENT_CREDITS')
      expect(body.error.details.available).toBe(10)
    })

    it('returns 409 when document is still processing', async () => {
      const user = await createTestUser()
      const plan = await createTestPlan()
      await createTestSubscription(user.id, plan.id)
      await grantTestCredits(user.id, 2000)
      const doc = await createTestDocument(user.id, { processingStatus: 'pending_embedding' })
      await createTestModelConfig('notes')

      const response = await request.post(
        '/study-sets/generate',
        { document_id: doc.id, types: ['notes'] },
        user.firebaseUid
      )

      expect(response.statusCode).toBe(409)
      const body = JSON.parse(response.body)
      expect(body.error.code).toBe('DOCUMENT_NOT_READY')
    })

    it('returns 404 when document belongs to another user', async () => {
      const user1 = await createTestUser()
      const user2 = await createTestUser()
      const plan = await createTestPlan()
      await createTestSubscription(user2.id, plan.id)
      await grantTestCredits(user2.id, 2000)
      const doc = await createTestDocument(user1.id)
      await createTestModelConfig('notes')

      const response = await request.post(
        '/study-sets/generate',
        { document_id: doc.id, types: ['notes'] },
        user2.firebaseUid
      )

      expect(response.statusCode).toBe(404)
    })

    it('returns 422 for invalid types array', async () => {
      const user = await createTestUser()
      const plan = await createTestPlan()
      await createTestSubscription(user.id, plan.id)
      await grantTestCredits(user.id, 2000)
      const doc = await createTestDocument(user.id)

      const response = await request.post(
        '/study-sets/generate',
        { document_id: doc.id, types: ['invalid_type'] },
        user.firebaseUid
      )

      expect(response.statusCode).toBe(422)
    })
  })
```

### `tests/unit/chunker.test.ts`

```tsx
import { describe, it, expect } from 'vitest'
import { Chunker } from '../../src/services/rag/chunker.js'

const longText = 'The quick brown fox jumps over the lazy dog. '.repeat(200)

describe('Chunker', () => {
  it('splits text into chunks that respect the max token size', () => {
    const chunks = Chunker.chunk(longText, { chunkSize: 100, overlap: 20 })
    expect(chunks.length).toBeGreaterThan(1)
    chunks.forEach(chunk => {
      expect(chunk.tokenCount).toBeLessThanOrEqual(110)
    })
  })

  it('produces overlapping content between consecutive chunks', () => {
    const chunks = Chunker.chunk(longText, { chunkSize: 100, overlap: 20 })
    if (chunks.length < 2) return
    const endOfFirst = chunks[0].text.slice(-50)
    const startOfSecond = chunks[1].text.slice(0, 50)
    const hasOverlap = startOfSecond.includes(endOfFirst.slice(0, 20))
    expect(hasOverlap).toBe(true)
  })

  it('returns a single chunk when text is shorter than chunkSize', () => {
    const shortText = 'Short text.'
    const chunks = Chunker.chunk(shortText, { chunkSize: 512, overlap: 64 })
    expect(chunks).toHaveLength(1)
    expect(chunks[0].text).toBe(shortText)
  })

  it('preserves all text across chunks with no dropped content', () => {
    const text = 'word '.repeat(1000).trim()
    const chunks = Chunker.chunk(text, { chunkSize: 128, overlap: 16 })
    const wordsInChunks = new Set(chunks.flatMap(c => c.text.split(' ')))
    const originalWords = new Set(text.split(' '))
    for (const word of originalWords) {
      expect(wordsInChunks.has(word)).toBe(true)
    }
  })
})
```
