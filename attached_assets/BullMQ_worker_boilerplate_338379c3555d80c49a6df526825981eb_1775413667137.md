### `src/workers/base.worker.ts`

```tsx
import { Worker, Job, UnrecoverableError } from 'bullmq'
import { db } from '../db/index.js'
import { bullmqJobs, jobRetryHistory, deadLetterJobs, jobEvents, workerHealth, studySetBatches, studySetOutputs } from '../db/schema/index.js'
import { LedgerService } from '../services/ledger/ledger.service.js'
import { wsEmitter } from '../websocket/ws.emitter.js'
import { logger } from '../utils/logger.js'
import { eq, sql } from 'drizzle-orm'
import { redis } from '../config/redis.js'
import { QUEUE_NAMES } from '../queues/queues.constants.js'

export interface WorkerJobData {
  jobId: string
  userId: string
  batchId: string
  documentId: string
  taskType: string
  idempotencyKey: string
}

export interface WorkerJobResult {
  outputData?: Record<string, unknown>
  r2Path?: string
  rawTokensUsed: number
  creditsConsumed: number
  modelName: string
  provider: string
}

export abstract class BaseWorker {
  protected worker: Worker
  private heartbeatInterval: ReturnType<typeof setInterval> | null = null
  private workerName: string
  private queueName: string

  constructor(queueName: string) {
    this.queueName = queueName
    this.workerName = `${queueName}-${process.pid}`

    this.worker = new Worker(
      queueName,
      async (job: Job<WorkerJobData>) => {
        return await this.handleJob(job)
      },
      {
        connection: redis,
        concurrency: parseInt(process.env.WORKER_CONCURRENCY ?? '3'),
        removeOnComplete: { count: 100 },
        removeOnFail: { count: 500 },
      }
    )

    this.worker.on('failed', async (job, err) => {
      if (job) await this.onJobFailed(job, err)
    })

    this.worker.on('error', (err) => {
      logger.error({ err, workerName: this.workerName }, 'Worker connection error')
    })

    this.startHeartbeat()
    this.registerShutdownHooks()

    logger.info({ workerName: this.workerName, queue: queueName }, 'Worker started')
  }

  private async handleJob(job: Job<WorkerJobData>): Promise<WorkerJobResult> {
    const { jobId, batchId, taskType } = job.data

    logger.info({ jobId, taskType, attempt: job.attemptsMade + 1 }, 'Job started')

    await db
      .update(bullmqJobs)
      .set({ status: 'processing', startedAt: new Date(), attemptNumber: job.attemptsMade + 1 })
      .where(eq(bullmqJobs.id, jobId))

    await this.emitEvent(batchId, { type: 'job_started', jobId, taskType, timestamp: new Date().toISOString() })

    await db.insert(jobEvents).values({
      jobId,
      eventType: 'started',
      metadata: { attempt: job.attemptsMade + 1 },
    })

    let result: WorkerJobResult

    try {
      result = await this.process(job)
    } catch (err) {
      const error = err as Error

      await db.insert(jobRetryHistory).values({
        jobId,
        attemptNumber: job.attemptsMade + 1,
        status: 'failed',
        errorMessage: error.message,
        attemptedAt: new Date(),
      })

      const maxAttempts = job.opts.attempts ?? 3
      const isLastAttempt = job.attemptsMade + 1 >= maxAttempts

      if (isLastAttempt) {
        throw new UnrecoverableError(error.message)
      }

      throw error
    }

    await this.onJobCompleted(job, result)
    return result
  }

  private async onJobCompleted(job: Job<WorkerJobData>, result: WorkerJobResult): Promise<void> {
    const { jobId, batchId, userId, taskType, idempotencyKey } = job.data

    await db.insert(studySetOutputs).values({
      batchId,
      userId,
      type: taskType,
      status: 'completed',
      outputData: result.outputData ?? null,
      r2Path: result.r2Path ?? null,
      modelName: result.modelName,
      provider: result.provider,
      rawTokensUsed: result.rawTokensUsed,
      creditsConsumed: result.creditsConsumed,
      completedAt: new Date(),
    }).returning()

    const [output] = await db
      .select({ id: studySetOutputs.id })
      .from(studySetOutputs)
      .where(eq(studySetOutputs.batchId, batchId))
      .orderBy(sql`created_at DESC`)
      .limit(1)

    await LedgerService.confirmDebit({
      userId,
      jobId,
      rawTokens: result.rawTokensUsed,
      multiplier: 1.0,
      computedCredits: result.creditsConsumed,
      taskType,
      modelName: result.modelName,
      provider: result.provider,
      idempotencyKey: `confirm_${idempotencyKey}`,
    })

    await db
      .update(bullmqJobs)
      .set({ status: 'completed', completedAt: new Date(), processingTimeMs: Date.now() - (job.processedOn ?? Date.now()) })
      .where(eq(bullmqJobs.id, jobId))

    await db
      .update(studySetBatches)
      .set({ completedJobs: sql`completed_jobs + 1` })
      .where(eq(studySetBatches.id, batchId))

    const [batch] = await db
      .select({ totalJobs: studySetBatches.totalJobs, completedJobs: studySetBatches.completedJobs, failedJobs: studySetBatches.failedJobs })
      .from(studySetBatches)
      .where(eq(studySetBatches.id, batchId))

    const isAllDone = batch.completedJobs + batch.failedJobs >= batch.totalJobs

    if (isAllDone) {
      const finalStatus = batch.failedJobs === batch.totalJobs ? 'failed' : batch.failedJobs > 0 ? 'partial' : 'completed'
      await db.update(studySetBatches).set({ status: finalStatus, completedAt: new Date() }).where(eq(studySetBatches.id, batchId))
      await this.emitEvent(batchId, {
        type: 'batch_completed',
        batchId,
        totalJobs: batch.totalJobs,
        completedJobs: batch.completedJobs,
        failedJobs: batch.failedJobs,
        timestamp: new Date().toISOString(),
      })
    }

    await db.insert(jobEvents).values({ jobId, eventType: 'completed', metadata: { creditsConsumed: result.creditsConsumed } })

    await this.emitEvent(batchId, {
      type: 'job_completed',
      jobId,
      taskType,
      outputId: output?.id,
      creditsConsumed: result.creditsConsumed,
      timestamp: new Date().toISOString(),
    })

    logger.info({ jobId, taskType, creditsConsumed: result.creditsConsumed }, 'Job completed')
  }

  private async onJobFailed(job: Job<WorkerJobData>, err: Error): Promise<void> {
    const { jobId, batchId, userId, taskType } = job.data

    logger.error({ jobId, taskType, err: err.message }, 'Job permanently failed')

    await db.insert(deadLetterJobs).values({
      originalJobId: jobId,
      userId,
      queueName: this.queueName,
      taskType,
      inputPayload: job.data as Record<string, unknown>,
      finalError: err.message,
      totalAttempts: job.attemptsMade,
      creditsReleased: false,
      requiresManualReview: true,
    })

    await LedgerService.releasePendingDebit({ userId, jobId, idempotencyKey: `release_${job.data.idempotencyKey}` })

    await db
      .update(bullmqJobs)
      .set({ status: 'dead', errorMessage: err.message })
      .where(eq(bullmqJobs.id, jobId))

    await db
      .update(studySetBatches)
      .set({ failedJobs: sql`failed_jobs + 1` })
      .where(eq(studySetBatches.id, batchId))

    await this.emitEvent(batchId, {
      type: 'job_failed',
      jobId,
      taskType,
      error: err.message,
      timestamp: new Date().toISOString(),
    })
  }

  private async emitEvent(batchId: string, event: Record<string, unknown>): Promise<void> {
    await wsEmitter.emit(`ws:batch:${batchId}`, event)
  }

  private startHeartbeat(): void {
    this.heartbeatInterval = setInterval(async () => {
      try {
        await db
          .insert(workerHealth)
          .values({
            workerName: this.workerName,
            queueName: this.queueName,
            host: process.env.HOSTNAME ?? 'unknown',
            pid: process.pid,
            status: 'healthy',
            lastHeartbeatAt: new Date(),
          })
          .onConflictDoUpdate({
            target: workerHealth.workerName,
            set: { status: 'healthy', lastHeartbeatAt: new Date() },
          })
      } catch (err) {
        logger.warn({ err }, 'Failed to write heartbeat')
      }
    }, 30_000)
  }

  private registerShutdownHooks(): void {
    const shutdown = async (signal: string) => {
      logger.info({ signal, workerName: this.workerName }, 'Shutting down worker gracefully')

      if (this.heartbeatInterval) clearInterval(this.heartbeatInterval)

      await db
        .update(workerHealth)
        .set({ status: 'stopped' })
        .where(eq(workerHealth.workerName, this.workerName))

      await this.worker.close()
      process.exit(0)
    }

    process.on('SIGTERM', () => shutdown('SIGTERM'))
    process.on('SIGINT', () => shutdown('SIGINT'))
  }

  abstract process(job: Job<WorkerJobData>): Promise<WorkerJobResult>

  async start(): Promise<void> {
    logger.info({ workerName: this.workerName }, 'Worker running and listening for jobs')
  }
}
```

### `src/workers/notes.worker.ts`

```tsx
import { Job } from 'bullmq'
import { BaseWorker, WorkerJobData, WorkerJobResult } from './base.worker.js'
import { RAGService } from '../services/rag/rag.service.js'
import { ModelRouter } from '../services/model-router/model-router.service.js'
import { PromptBuilder } from '../services/prompt-builder/prompt-builder.service.js'
import { db } from '../db/index.js'
import { outputCache } from '../db/schema/index.js'
import { and, eq } from 'drizzle-orm'
import { QUEUE_NAMES } from '../queues/queues.constants.js'
import { logger } from '../utils/logger.js'

export class NotesWorker extends BaseWorker {
  constructor() {
    super(QUEUE_NAMES.NOTES)
  }

  async process(job: Job<WorkerJobData>): Promise<WorkerJobResult> {
    const { documentId, userId, jobId, taskType } = job.data

    const [cached] = await db
      .select()
      .from(outputCache)
      .where(and(eq(outputCache.taskType, taskType), eq(outputCache.contentHash, documentId)))
      .limit(1)

    if (cached) {
      logger.info({ jobId, taskType }, 'Cache hit — returning cached output')
      await db.update(outputCache).set({ hitCount: cached.hitCount + 1 }).where(eq(outputCache.id, cached.id))
      return {
        outputData: cached.outputData as Record<string, unknown>,
        rawTokensUsed: 0,
        creditsConsumed: 0,
        modelName: cached.modelName,
        provider: 'cache',
      }
    }

    const context = await RAGService.search({
      documentId,
      taskType,
      topK: 10,
      userId,
      jobId,
    })

    const prompt = await PromptBuilder.build({ taskType, context })

    const response = await ModelRouter.generate({
      taskType,
      prompt,
      userId,
      jobId,
      maxTokens: 4096,
    })

    const outputData = { format: 'markdown', markdown: response.output }

    await db.insert(outputCache).values({
      contentHash: documentId,
      taskType,
      modelName: response.modelName,
      outputData,
      expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000),
    }).onConflictDoNothing()

    return {
      outputData,
      rawTokensUsed: response.rawTokens,
      creditsConsumed: response.computedCredits,
      modelName: response.modelName,
      provider: response.provider,
    }
  }
}
```

### `src/workers/podcast.worker.ts`

```tsx
import { Job } from 'bullmq'
import { BaseWorker, WorkerJobData, WorkerJobResult } from './base.worker.js'
import { RAGService } from '../services/rag/rag.service.js'
import { ModelRouter } from '../services/model-router/model-router.service.js'
import { PromptBuilder } from '../services/prompt-builder/prompt-builder.service.js'
import { TTSService } from '../services/tts/tts.service.js'
import { StorageService } from '../services/storage/storage.service.js'
import { QUEUE_NAMES } from '../queues/queues.constants.js'
import { logger } from '../utils/logger.js'
import { randomUUID } from 'crypto'

export class PodcastWorker extends BaseWorker {
  constructor() {
    super(QUEUE_NAMES.PODCAST)
  }

  async process(job: Job<WorkerJobData>): Promise<WorkerJobResult> {
    const { documentId, userId, jobId, taskType } = job.data

    logger.info({ jobId }, 'Generating podcast script')

    const context = await RAGService.search({ documentId, taskType, topK: 14, userId, jobId })
    const prompt = await PromptBuilder.build({ taskType, context })

    const scriptResponse = await ModelRouter.generate({
      taskType: 'podcast_script',
      prompt,
      userId,
      jobId,
      maxTokens: 6000,
    })

    logger.info({ jobId }, 'Synthesizing audio from script')

    const audioBuffer = await TTSService.synthesize({
      script: scriptResponse.output,
      voiceA: process.env.OPENAI_TTS_VOICE_A ?? 'alloy',
      voiceB: process.env.OPENAI_TTS_VOICE_B ?? 'echo',
    })

    const outputId = randomUUID()
    const r2Path = `podcasts/${userId}/${outputId}.mp3`

    await StorageService.upload({
      key: r2Path,
      body: audioBuffer,
      contentType: 'audio/mpeg',
    })

    logger.info({ jobId, r2Path }, 'Podcast audio uploaded to R2')

    const scriptLines = scriptResponse.output.split('\n').slice(0, 4).join('\n')

    return {
      outputData: { scriptPreview: scriptLines, durationNote: 'See r2_path for audio' },
      r2Path,
      rawTokensUsed: scriptResponse.rawTokens,
      creditsConsumed: scriptResponse.computedCredits,
      modelName: scriptResponse.modelName,
      provider: scriptResponse.provider,
    }
  }
}
```

### `src/workers/embedding.worker.ts`

```tsx
import { Job } from 'bullmq'
import { BaseWorker, WorkerJobData, WorkerJobResult } from './base.worker.js'
import { db } from '../db/index.js'
import { documents, documentChunks, embeddingJobs } from '../db/schema/index.js'
import { Chunker } from '../services/rag/chunker.js'
import { ModelRouter } from '../services/model-router/model-router.service.js'
import { eq } from 'drizzle-orm'
import { QUEUE_NAMES } from '../queues/queues.constants.js'
import { logger } from '../utils/logger.js'

export class EmbeddingWorker extends BaseWorker {
  constructor() {
    super(QUEUE_NAMES.EMBEDDING)
  }

  async process(job: Job<WorkerJobData>): Promise<WorkerJobResult> {
    const { documentId, jobId } = job.data

    const [document] = await db.select().from(documents).where(eq(documents.id, documentId))

    if (!document?.extractedText) {
      throw new Error(`Document ${documentId} has no extracted text`)
    }

    await db.update(documents).set({ processingStatus: 'embedding' }).where(eq(documents.id, documentId))

    await db.insert(embeddingJobs).values({
      documentId,
      status: 'processing',
      embeddingModel: 'text-embedding-3-small',
      embeddingProvider: 'openai',
    })

    const chunks = Chunker.chunk(document.extractedText, { chunkSize: 512, overlap: 64 })

    await db.delete(documentChunks).where(eq(documentChunks.documentId, documentId))

    let totalTokens = 0

    for (let i = 0; i < chunks.length; i++) {
      const chunk = chunks[i]

      const embeddingResponse = await ModelRouter.embed({
        text: chunk.text,
        jobId,
        userId: job.data.userId,
      })

      await db.insert(documentChunks).values({
        documentId,
        chunkIndex: i,
        chunkText: chunk.text,
        tokenCount: chunk.tokenCount,
        embedding: embeddingResponse.vector,
      })

      totalTokens += embeddingResponse.rawTokens

      await db
        .update(embeddingJobs)
        .set({ processedChunks: i + 1 })
        .where(eq(embeddingJobs.documentId, documentId))

      logger.debug({ documentId, chunk: i + 1, total: chunks.length }, 'Chunk embedded')
    }

    await db.update(embeddingJobs).set({ status: 'completed', tokensUsed: totalTokens, completedAt: new Date() }).where(eq(embeddingJobs.documentId, documentId))
    await db.update(documents).set({ processingStatus: 'ready' }).where(eq(documents.id, documentId))

    logger.info({ documentId, totalChunks: chunks.length, totalTokens }, 'Embedding complete')

    return {
      rawTokensUsed: totalTokens,
      creditsConsumed: Math.ceil(totalTokens * 0.1),
      modelName: 'text-embedding-3-small',
      provider: 'openai',
    }
  }
}
```

### `src/workers/index.ts`

```tsx
import { logger } from '../utils/logger.js'

const WORKER_MAP = {
  notes: () => import('./notes.worker.js').then(m => new m.NotesWorker()),
  flashcards: () => import('./flashcards.worker.js').then(m => new m.FlashcardsWorker()),
  multiple_choice: () => import('./multiple-choice.worker.js').then(m => new m.MultipleChoiceWorker()),
  fill_in_blanks: () => import('./fill-in-blanks.worker.js').then(m => new m.FillInBlanksWorker()),
  written_test: () => import('./written-test.worker.js').then(m => new m.WrittenTestWorker()),
  tutor_lesson: () => import('./tutor-lesson.worker.js').then(m => new m.TutorLessonWorker()),
  content: () => import('./content.worker.js').then(m => new m.ContentWorker()),
  podcast: () => import('./podcast.worker.js').then(m => new m.PodcastWorker()),
  embedding: () => import('./embedding.worker.js').then(m => new m.EmbeddingWorker()),
  grader: () => import('./grader.worker.js').then(m => new m.GraderWorker()),
  scheduler: () => import('./scheduler.worker.js').then(m => new m.SchedulerWorker()),
}

async function main() {
  const workerType = process.env.WORKER_TYPE

  if (!workerType || !(workerType in WORKER_MAP)) {
    logger.error({ workerType, available: Object.keys(WORKER_MAP) }, 'Invalid or missing WORKER_TYPE')
    process.exit(1)
  }

  const factory = WORKER_MAP[workerType as keyof typeof WORKER_MAP]
  const worker = await factory()
  await worker.start()

  process.on('uncaughtException', (err) => {
    logger.error({ err }, 'Uncaught exception in worker process')
    process.exit(1)
  })

  process.on('unhandledRejection', (reason) => {
    logger.error({ reason }, 'Unhandled rejection in worker process')
    process.exit(1)
  })
}

main()
```
