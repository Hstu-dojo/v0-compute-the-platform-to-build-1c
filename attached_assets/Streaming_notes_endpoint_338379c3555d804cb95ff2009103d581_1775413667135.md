### `src/services/model-router/model-router.service.ts`

```tsx
import OpenAI from 'openai'
import Anthropic from '@anthropic-ai/sdk'
import { db } from '../../db/index.js'
import { modelConfig, aiRequestLogs } from '../../db/schema/index.js'
import { and, eq } from 'drizzle-orm'
import { logger } from '../../utils/logger.js'

export interface ModelRequest {
  taskType: string
  prompt: { system: string; user: string }
  userId: string
  jobId: string
  maxTokens?: number
}

export interface ModelStreamRequest extends ModelRequest {
  onDelta: (delta: string) => void | Promise<void>
}

export interface ModelResponse {
  output: string
  rawTokens: number
  computedCredits: number
  modelName: string
  provider: string
  latencyMs: number
}

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })
const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY })

export class ModelRouter {
  static async generate(req: ModelRequest): Promise<ModelResponse> {
    const config = await this.resolveConfig(req.taskType, req.userId)
    const start = Date.now()

    try {
      const result = await this.callProvider(config.provider, config.modelName, req)
      const latencyMs = Date.now() - start
      const computedCredits = Math.ceil(result.rawTokens * config.tokenMultiplier)

      await this.logRequest({ ...req, ...config, ...result, latencyMs, status: 'success' })

      return { ...result, computedCredits, modelName: config.modelName, provider: config.provider, latencyMs }
    } catch (err) {
      if (config.fallbackModelName && config.fallbackProvider) {
        logger.warn({ taskType: req.taskType, err }, 'Primary provider failed — using fallback')
        const fallbackConfig = { ...config, modelName: config.fallbackModelName, provider: config.fallbackProvider }
        const result = await this.callProvider(fallbackConfig.provider, fallbackConfig.modelName, req)
        const latencyMs = Date.now() - start
        const computedCredits = Math.ceil(result.rawTokens * config.tokenMultiplier)
        await this.logRequest({ ...req, ...fallbackConfig, ...result, latencyMs, status: 'fallback_used' })
        return { ...result, computedCredits, modelName: fallbackConfig.modelName, provider: fallbackConfig.provider, latencyMs }
      }
      throw err
    }
  }

  static async stream(req: ModelStreamRequest): Promise<ModelResponse> {
    const config = await this.resolveConfig(req.taskType, req.userId)
    const start = Date.now()
    let fullOutput = ''
    let promptTokens = 0
    let completionTokens = 0

    if (config.provider === 'openai') {
      const stream = openai.chat.completions.stream({
        model: config.modelName,
        max_tokens: req.maxTokens ?? 4096,
        messages: [
          { role: 'system', content: req.prompt.system },
          { role: 'user', content: req.prompt.user },
        ],
      })

      for await (const chunk of stream) {
        const delta = chunk.choices[0]?.delta?.content ?? ''
        if (delta) {
          fullOutput += delta
          await req.onDelta(delta)
        }
      }

      const final = await stream.finalChatCompletion()
      promptTokens = final.usage?.prompt_tokens ?? 0
      completionTokens = final.usage?.completion_tokens ?? 0
    } else if (config.provider === 'anthropic') {
      const stream = anthropic.messages.stream({
        model: config.modelName,
        max_tokens: req.maxTokens ?? 4096,
        system: req.prompt.system,
        messages: [{ role: 'user', content: req.prompt.user }],
      })

      for await (const event of stream) {
        if (event.type === 'content_block_delta' && event.delta.type === 'text_delta') {
          const delta = event.delta.text
          fullOutput += delta
          await req.onDelta(delta)
        }
      }

      const final = await stream.finalMessage()
      promptTokens = final.usage.input_tokens
      completionTokens = final.usage.output_tokens
    } else {
      throw new Error(`Provider ${config.provider} does not support streaming`)
    }

    const rawTokens = promptTokens + completionTokens
    const computedCredits = Math.ceil(rawTokens * config.tokenMultiplier)
    const latencyMs = Date.now() - start

    await this.logRequest({
      ...req,
      ...config,
      rawTokens,
      latencyMs,
      status: 'success',
      output: fullOutput,
    })

    return { output: fullOutput, rawTokens, computedCredits, modelName: config.modelName, provider: config.provider, latencyMs }
  }

  static async embed(req: { text: string; jobId: string; userId: string }): Promise<{ vector: number[]; rawTokens: number }> {
    const response = await openai.embeddings.create({
      model: 'text-embedding-3-small',
      input: req.text,
    })

    return {
      vector: response.data[0].embedding,
      rawTokens: response.usage.total_tokens,
    }
  }

  private static async callProvider(
    provider: string,
    modelName: string,
    req: ModelRequest
  ): Promise<{ output: string; rawTokens: number }> {
    if (provider === 'openai') {
      const response = await openai.chat.completions.create({
        model: modelName,
        max_tokens: req.maxTokens ?? 2048,
        messages: [
          { role: 'system', content: req.prompt.system },
          { role: 'user', content: req.prompt.user },
        ],
      })
      return {
        output: response.choices[0]?.message?.content ?? '',
        rawTokens: response.usage?.total_tokens ?? 0,
      }
    }

    if (provider === 'anthropic') {
      const response = await anthropic.messages.create({
        model: modelName,
        max_tokens: req.maxTokens ?? 2048,
        system: req.prompt.system,
        messages: [{ role: 'user', content: req.prompt.user }],
      })
      return {
        output: response.content[0].type === 'text' ? response.content[0].text : '',
        rawTokens: response.usage.input_tokens + response.usage.output_tokens,
      }
    }

    throw new Error(`Unknown provider: ${provider}`)
  }

  private static async resolveConfig(taskType: string, userId: string) {
    const configs = await db
      .select()
      .from(modelConfig)
      .where(and(eq(modelConfig.taskType, taskType), eq(modelConfig.isActive, true)))

    if (!configs.length) throw new Error(`No active model config for task type: ${taskType}`)

    const config = configs[0]

    if (config.rolloutPercentage < 100) {
      const hash = userId.split('').reduce((acc, c) => acc + c.charCodeAt(0), 0) % 100
      if (hash >= config.rolloutPercentage) {
        const stable = configs.find(c => c.rolloutPercentage === 100)
        if (stable) return stable
      }
    }

    return config
  }

  private static async logRequest(data: Record<string, unknown>): Promise<void> {
    try {
      await db.insert(aiRequestLogs).values({
        userId: data.userId as string,
        jobId: data.jobId as string,
        taskType: data.taskType as string,
        provider: data.provider as string,
        modelName: data.modelName as string,
        systemPrompt: (data.prompt as { system: string }).system,
        userPrompt: (data.prompt as { user: string }).user,
        promptTokens: data.promptTokens as number ?? 0,
        completionTokens: data.completionTokens as number ?? 0,
        totalTokens: data.rawTokens as number ?? 0,
        latencyMs: data.latencyMs as number,
        status: data.status as string,
      })
    } catch (err) {
      logger.warn({ err }, 'Failed to write ai_request_log — non-fatal')
    }
  }
}
```

### `src/modules/outputs/outputs.stream.ts`

```tsx
import { FastifyRequest, FastifyReply } from 'fastify'
import { db } from '../../db/index.js'
import { studySetOutputs, documents, studySetBatches } from '../../db/schema/index.js'
import { and, eq } from 'drizzle-orm'
import { ModelRouter } from '../../services/model-router/model-router.service.js'
import { RAGService } from '../../services/rag/rag.service.js'
import { PromptBuilder } from '../../services/prompt-builder/prompt-builder.service.js'
import { LedgerService } from '../../services/ledger/ledger.service.js'
import { logger } from '../../utils/logger.js'
import { randomUUID } from 'crypto'

interface StreamParams {
  output_id: string
}

export async function streamNotesHandler(
  request: FastifyRequest<{ Params: StreamParams }>,
  reply: FastifyReply
): Promise<void> {
  const { output_id } = request.params
  const userId = request.user.id

  const [output] = await db
    .select()
    .from(studySetOutputs)
    .where(and(eq(studySetOutputs.id, output_id), eq(studySetOutputs.userId, userId)))

  if (!output) {
    reply.code(404).send({ error: { code: 'NOT_FOUND', message: 'Output not found.' } })
    return
  }

  reply.raw.setHeader('Content-Type', 'text/event-stream')
  reply.raw.setHeader('Cache-Control', 'no-cache')
  reply.raw.setHeader('Connection', 'keep-alive')
  reply.raw.setHeader('X-Accel-Buffering', 'no')
  reply.raw.flushHeaders()

  const send = (event: string, data: unknown) => {
    reply.raw.write(`event: ${event}\ndata: ${JSON.stringify(data)}\n\n`)
  }

  const closeStream = () => {
    reply.raw.end()
  }

  request.raw.on('close', () => {
    logger.debug({ output_id }, 'Client disconnected from SSE stream')
  })

  if (output.status === 'completed' && output.outputData) {
    const markdown = (output.outputData as { markdown: string }).markdown

    send('metadata', { output_id, model: output.modelName, provider: output.provider })

    const chunkSize = 100
    for (let i = 0; i < markdown.length; i += chunkSize) {
      send('delta', { delta: markdown.slice(i, i + chunkSize) })
      await sleep(5)
    }

    send('done', { creditsConsumed: output.creditsConsumed, cached: true })
    closeStream()
    return
  }

  if (output.status === 'processing' || output.status === 'queued') {
    send('metadata', { output_id, model: null, provider: null })
    send('error', { code: 'NOT_READY', message: 'This output is still being generated. Use the WebSocket to wait for completion, then stream.' })
    closeStream()
    return
  }

  const [batch] = await db.select().from(studySetBatches).where(eq(studySetBatches.id, output.batchId))
  const documentId = batch.documentId

  send('metadata', { output_id, generating: true })

  let fullMarkdown = ''
  let streamError: Error | null = null

  const idempotencyKey = `stream_${output_id}_${randomUUID()}`

  await LedgerService.insertPendingDebit({
    userId,
    jobId: output_id,
    estimatedCredits: 350,
    taskType: 'notes',
    idempotencyKey: `pending_${idempotencyKey}`,
  })

  try {
    const context = await RAGService.search({
      documentId,
      taskType: 'notes',
      topK: 10,
      userId,
      jobId: output_id,
    })

    const prompt = await PromptBuilder.build({ taskType: 'notes', context })

    const result = await ModelRouter.stream({
      taskType: 'notes',
      prompt,
      userId,
      jobId: output_id,
      maxTokens: 4096,
      onDelta: async (delta: string) => {
        fullMarkdown += delta
        send('delta', { delta })
      },
    })

    await db
      .update(studySetOutputs)
      .set({
        status: 'completed',
        outputData: { format: 'markdown', markdown: fullMarkdown },
        modelName: result.modelName,
        provider: result.provider,
        rawTokensUsed: result.rawTokens,
        creditsConsumed: result.computedCredits,
        completedAt: new Date(),
      })
      .where(eq(studySetOutputs.id, output_id))

    await LedgerService.confirmDebit({
      userId,
      jobId: output_id,
      rawTokens: result.rawTokens,
      multiplier: 1.0,
      computedCredits: result.computedCredits,
      taskType: 'notes',
      modelName: result.modelName,
      provider: result.provider,
      idempotencyKey: `confirm_${idempotencyKey}`,
    })

    send('done', {
      creditsConsumed: result.computedCredits,
      promptTokens: result.rawTokens,
      model: result.modelName,
    })
  } catch (err) {
    streamError = err as Error
    logger.error({ err, output_id }, 'Stream failed')

    await LedgerService.releasePendingDebit({
      userId,
      jobId: output_id,
      idempotencyKey: `release_${idempotencyKey}`,
    })

    await db
      .update(studySetOutputs)
      .set({ status: 'failed', errorMessage: streamError.message })
      .where(eq(studySetOutputs.id, output_id))

    send('error', { code: 'STREAM_INTERRUPTED', message: 'Connection to AI provider lost mid-stream.' })
  } finally {
    closeStream()
  }
}

function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms))
}
```

### `src/modules/outputs/outputs.routes.ts`

```tsx
import { FastifyInstance } from 'fastify'
import { streamNotesHandler } from './outputs.stream.js'
import { OutputsController } from './outputs.controller.js'
import { authMiddleware } from '../../middleware/auth.middleware.js'
import { hydrateMiddleware } from '../../middleware/hydrate.middleware.js'

export async function outputsRoutes(app: FastifyInstance): Promise<void> {
  app.addHook('preHandler', authMiddleware)
  app.addHook('preHandler', hydrateMiddleware)

  app.get('/study-sets/output/:output_id/notes', OutputsController.getNotes)
  app.get('/study-sets/output/:output_id/notes/stream', streamNotesHandler)
  app.get('/study-sets/output/:output_id/flashcards', OutputsController.getFlashcards)
  app.get('/study-sets/output/:output_id/multiple_choice', OutputsController.getMultipleChoice)
  app.get('/study-sets/output/:output_id/fill_in_blanks', OutputsController.getFillInBlanks)
  app.get('/study-sets/output/:output_id/written_test', OutputsController.getWrittenTest)
  app.get('/study-sets/output/:output_id/tutor_lesson', OutputsController.getTutorLesson)
  app.get('/study-sets/output/:output_id/podcast', OutputsController.getPodcast)
  app.get('/study-sets/output/:output_id/content', OutputsController.getContent)
}
```
